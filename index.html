npx

# 初始化，并选择 2.x
npx @vuepress-reco/theme-cli init

npm

# 初始化，并选择 2.x
npm install @vuepress-reco/theme-cli@1.0.7 -g
theme-cli init

yarn

# 初始化，并选择 2.x
yarn global add @vuepress-reco/theme-cli@1.0.7
theme-cli init
