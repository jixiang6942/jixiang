# 1，防抖(debounce)函数的手写

什么是防抖？在一段时间内再次触发某个事件，这个事件会等到一个新的周期结束后执行。类似于英雄联盟中的回城。

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
</head>
<body>
  
  <input type="text">
  <script>
    // 手写防抖函数
    function debounce(fn, delay = 1000) {
      let timer = null

      return (event) => {
        if (timer) {
          clearTimeout(timer)
        }

        timer = setTimeout(() => {
          fn()
        }, delay);
      }
    }

    // 1. 获取input元素
    const iptEl = document.querySelector('input')

    // 2. 给input元素添加输入事件
    iptEl.addEventListener('input', debounce(() => {
      console.log('有人输入了内容~~~')
    }, 2000))
  </script>
</body>
</html>

    // 2.手动实现节流(throttle)效果
    // let flag = true
    // btnEl.onclick = function() {
    //   if (!flag) return 
    //   flag = false
    //   console.log('被点击了~~~~')

    //   setTimeout(() => {
    //     flag = true
    //   }, 1000);
    // }

    btnEl.onclick = throttle(() => {
      console.log('点我有惊喜~~~~')
    }, 1000)

    // 3. 节流(throttle)函数
    function throttle(fn, delay) {
      let flag = true
      
      return function() {
        // 1.一进入就要关闭阀门
        if (!flag) return 
        flag = false
        fn()

        setTimeout(() => {
          // 2.时间到了之后，就打开阀门
          flag = true
        }, delay);
      }
    }
  </script>
</body>
</html>

Array.prototype.myFilter = function(fn) {
  let arr = []
  for(let i = 0; i < this.length; i++) {
    if (fn(this[i], i)) {
      arr[arr.length] = this[i]
    }
  }
  return arr
}

# F）call方法


// 手写一个call方法
Function.prototype.myCall = function(thisArg, ...args) {
  // 将方法放在对象thisArg身上, 通过隐式绑定实现call方法

  // 1. 将调用myCall的函数, 作为对象thisArg上的一个属性
  thisArg[this.name] = this

  // 2. 再通过隐式绑定调用函数, 因此调用call方法的函数中的this指向了thisArg
  return thisArg[this.name](...args)


}

// 作为this指向的对象
const obj = {

# 4，手写闭包



// 闭包的概念: 函数的局部变量默认是不能被函数外访问, 但是return一个子函数, 并且在子函数中中将该局部变量return出去. 
//  
// 闭包的作用:  延伸了变量的作用范围
// 闭包的场景:  手写的防抖函数
// 闭包的影响:  因为子函数内部一直引用着父函数的局部变量. 因此父函数执行完成后, 
// 父函数的局部变量不会被内存回收(js内存回收)


function closure() {
  let num = 10

  return function() {
    return num
  }
}

// 在外部使用函数内部的局部变量会报错
// console.log(num) // 报错

// 通过闭包可以实现
const fn = closure()
console.log(fn()) // 10

import { PureComponent } from 'react'
import { StoreContext } from './StoreContext'

// connect的参数:
// 参数一：函数
// 参数二：函数
// 返回值：函数

export function connect(mapStateToProps, mapDispatchToProps) {
  // 高阶组件：函数
  return function (WrapperComponent) {
    class NewComponent extends PureComponent {
      constructor(props, context) {
        super(props)

        this.state = mapStateToProps(context.getState())
      }

      componentDidMount() {
        this.unsubscribe = this.context.subscribe(() => {
          this.setState(mapStateToProps(this.context.getState()))
        })
      }

      componentWillUnmount() {
        this.unsubscribe()
      }

      render() {
        return (
          <WrapperComponent
            {...this.props}
            {...mapStateToProps(this.context.getState())}
            {...mapDispatchToProps(this.context.dispatch)}
          />
        )
      }
    }

    NewComponent.displayName = WrapperComponent.name
    NewComponent.contextType = StoreContext
    return NewComponent
  }
}
